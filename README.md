## Installation

```shell
npm install --save react-animated-dots

# OR

yarn add react-animated-dots
```

## Usage
In your `.js/jsx` file:

```javascript
import React from 'react';
import { Dot } from 'react-animated-dots';

const App = () => {
  return (
    <div>
      <h2>
      // Each <Dot> is a <span> in itself
      // You can include as much <Dot> as you want
        <Dot>.</Dot>
        <Dot>.</Dot>
        <Dot>.</Dot>
      </h2>
    </div>
  );
};

export default App;
```
